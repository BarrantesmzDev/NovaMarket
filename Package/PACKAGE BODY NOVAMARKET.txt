    FUNCTION FIDE_FACTURAS_CALCULAR_VENTAS_SUCURSAL_FN(
        p_id_sucursal IN NUMBER,
        p_mes IN NUMBER,
        p_anio IN NUMBER
    ) RETURN NUMBER
    IS
        v_total_ventas NUMBER := 0;
    BEGIN
        SELECT NVL(SUM(TOTAL), 0)
        INTO v_total_ventas
        FROM FIDE_FACTURAS_TB
        WHERE FIDE_FACTURAS_ID_SUCURSAL_FK = p_id_sucursal
        AND EXTRACT(MONTH FROM FECHA_FACTURA) = p_mes
        AND EXTRACT(YEAR FROM FECHA_FACTURA) = p_anio
        AND FIDE_FACTURAS_ID_ESTADO_FK = 1;
        
        RETURN v_total_ventas;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RETURN 0;
        WHEN OTHERS THEN
            RETURN 0;
    END FIDE_FACTURAS_CALCULAR_VENTAS_SUCURSAL_FN;

    -- 3. Función para verificar stock disponible
    FUNCTION FIDE_INVENTARIOS_VERIFICAR_STOCK_FN(
        p_id_producto IN NUMBER,
        p_id_sucursal IN NUMBER
    ) RETURN VARCHAR2
    IS
        v_stock_actual NUMBER;
        v_stock_minimo NUMBER;
        v_resultado VARCHAR2(50);
    BEGIN
        SELECT CANTIDAD_STOCK, CANTIDAD_MINIMA
        INTO v_stock_actual, v_stock_minimo
        FROM FIDE_INVENTARIOS_TB
        WHERE FIDE_INVENTARIOS_ID_PRODUCTO_FK = p_id_producto
        AND FIDE_INVENTARIOS_ID_SUCURSAL_FK = p_id_sucursal
        AND FIDE_INVENTARIOS_ID_ESTADO_FK = 1;
        
        IF v_stock_actual <= 0 THEN
            v_resultado := 'SIN_STOCK';
        ELSIF v_stock_actual <= v_stock_minimo THEN
            v_resultado := 'STOCK_BAJO';
        ELSE
            v_resultado := 'STOCK_SUFICIENTE';
        END IF;
        
        RETURN v_resultado;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RETURN 'NO_EXISTE';
        WHEN OTHERS THEN
            RETURN 'ERROR';
    END FIDE_INVENTARIOS_VERIFICAR_STOCK_FN;

    -- 4. Función para calcular salario neto empleado
    FUNCTION FIDE_PLANILLAS_CALCULAR_SALARIO_NETO_FN(
        p_id_empleado IN NUMBER,
        p_mes IN NUMBER,
        p_anio IN NUMBER
    ) RETURN NUMBER
    IS
        v_salario_bruto NUMBER := 0;
        v_deducciones NUMBER := 0;
        v_salario_neto NUMBER := 0;
    BEGIN
        SELECT SALARIO_BRUTO, DEDUCCIONES
        INTO v_salario_bruto, v_deducciones
        FROM FIDE_PLANILLAS_TB
        WHERE FIDE_PLANILLAS_ID_EMPLEADO_FK = p_id_empleado
        AND MES = p_mes
        AND ANIO = p_anio
        AND FIDE_PLANILLAS_ID_ESTADO_FK = 1;
        
        v_salario_neto := v_salario_bruto - v_deducciones;
        RETURN GREATEST(v_salario_neto, 0);
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RETURN 0;
        WHEN OTHERS THEN
            RETURN 0;
    END FIDE_PLANILLAS_CALCULAR_SALARIO_NETO_FN;

    -- 6. Función para validar credenciales de usuario
    FUNCTION FIDE_USUARIOS_VALIDAR_CREDENCIALES_FN(
        p_usuario IN VARCHAR2,
        p_contrasena IN VARCHAR2
    ) RETURN NUMBER
    IS
        v_id_usuario NUMBER := 0;
    BEGIN
        SELECT FIDE_USUARIOS_ID_USUARIO_PK
        INTO v_id_usuario
        FROM FIDE_USUARIOS_TB
        WHERE (CEDULA = p_usuario OR 
              FIDE_USUARIOS_ID_USUARIO_PK IN (
                  SELECT FIDE_USUARIOS_ID_USUARIO_PK 
                  FROM FIDE_USUARIOS_TB u
                  JOIN FIDE_CORREOS_TB c ON u.FIDE_USUARIOS_ID_CORREO_FK = c.FIDE_CORREOS_ID_CORREO_PK
                  WHERE c.CORREO = p_usuario
              ))
        AND CONTRASENA = p_contrasena
        AND FIDE_USUARIOS_ID_ESTADO_FK = 1
        AND ROWNUM = 1;
        
        RETURN v_id_usuario;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RETURN 0;
        WHEN OTHERS THEN
            RETURN 0;
    END FIDE_USUARIOS_VALIDAR_CREDENCIALES_FN;

    -- 11. Función para calcular descuento aplicable
    FUNCTION FIDE_DESCUENTOS_CALCULAR_APLICABLE_FN(
        p_id_cliente IN NUMBER,
        p_monto_compra IN NUMBER
    ) RETURN NUMBER
    IS
        v_descuento_total NUMBER := 0;
        v_porcentaje NUMBER := 0;
    BEGIN
        SELECT MAX(d.PORCENTAJE)
        INTO v_porcentaje
        FROM FIDE_DESCUENTOS_TB d
        JOIN FIDE_CLIENTES_DESCUENTO_TB cd ON d.FIDE_DESCUENTOS_ID_DESCUENTO_PK = cd.FIDE_CLIENTES_DESCUENTO_ID_DESCUENTO_FK
        WHERE cd.FIDE_CLIENTES_DESCUENTO_ID_USUARIO_FK = p_id_cliente
        AND SYSDATE BETWEEN d.FECHA_INICIO AND d.FECHA_FIN
        AND cd.FIDE_CLIENTES_DESCUENTO_ID_ESTADO_FK = 1
        AND d.FIDE_DESCUENTOS_ID_ESTADO_FK = 1;
        
        v_descuento_total := (p_monto_compra * NVL(v_porcentaje, 0)) / 100;
        RETURN ROUND(v_descuento_total, 2);
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RETURN 0;
        WHEN OTHERS THEN
            RETURN 0;
    END FIDE_DESCUENTOS_CALCULAR_APLICABLE_FN;

    -- 13. Función para calcular valor total del inventario
    FUNCTION FIDE_INVENTARIOS_CALCULAR_VALOR_TOTAL_FN(
        p_id_sucursal IN NUMBER
    ) RETURN NUMBER
    IS
        v_valor_total NUMBER := 0;
    BEGIN
        SELECT NVL(SUM(i.CANTIDAD_STOCK * p.PRECIO_UNITARIO), 0)
        INTO v_valor_total
        FROM FIDE_INVENTARIOS_TB i
        JOIN FIDE_PRODUCTOS_TB p ON i.FIDE_INVENTARIOS_ID_PRODUCTO_FK = p.FIDE_PRODUCTOS_ID_PRODUCTO_PK
        WHERE i.FIDE_INVENTARIOS_ID_SUCURSAL_FK = p_id_sucursal
        AND i.FIDE_INVENTARIOS_ID_ESTADO_FK = 1
        AND p.FIDE_PRODUCTOS_ID_ESTADO_FK = 1;
        
        RETURN ROUND(v_valor_total, 2);
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RETURN 0;
        WHEN OTHERS THEN
            RETURN 0;
    END FIDE_INVENTARIOS_CALCULAR_VALOR_TOTAL_FN;

    -- 15. Función para calcular promedio ventas por cliente
    FUNCTION FIDE_FACTURAS_CALCULAR_PROMEDIO_VENTAS_CLIENTE_FN(
        p_id_cliente IN NUMBER
    ) RETURN NUMBER
    IS
        v_total_ventas NUMBER := 0;
        v_cantidad_ventas NUMBER := 0;
        v_promedio NUMBER := 0;
    BEGIN
        SELECT NVL(SUM(TOTAL), 0), COUNT(*)
        INTO v_total_ventas, v_cantidad_ventas
        FROM FIDE_FACTURAS_TB
        WHERE FIDE_FACTURAS_ID_CLIENTE_FK = p_id_cliente
        AND FIDE_FACTURAS_ID_ESTADO_FK = 1;
        
        IF v_cantidad_ventas > 0 THEN
            v_promedio := v_total_ventas / v_cantidad_ventas;
        END IF;
        
        RETURN ROUND(v_promedio, 2);
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RETURN 0;
        WHEN OTHERS THEN
            RETURN 0;
    END FIDE_FACTURAS_CALCULAR_PROMEDIO_VENTAS_CLIENTE_FN;

  PROCEDURE FIDE_USUARIOS_LOGIN_USUARIO_SP(
        P_USUARIO IN VARCHAR2,
        P_RESULT  OUT VARCHAR2
    )
    IS
        V_ID_USUARIO   NUMBER;
        V_NOMBRE       VARCHAR2(200);
        V_TIPO_USUARIO VARCHAR2(50);
    BEGIN
        -----------------------------------------------------------------
        -- 1. BUSCAR USUARIO POR CORREO
        -----------------------------------------------------------------
        SELECT 
            U.FIDE_USUARIOS_ID_USUARIO_PK,
            U.NOMBRE || ' ' || U.APELLIDO_PATERNO || ' ' || U.APELLIDO_MATERNO,
            U.TIPO_USUARIO
        INTO V_ID_USUARIO, V_NOMBRE, V_TIPO_USUARIO
        FROM FIDE_USUARIOS_TB U
        JOIN FIDE_CORREOS_TB C
            ON U.FIDE_USUARIOS_ID_CORREO_FK = C.FIDE_CORREOS_ID_CORREO_PK
        WHERE TRIM(UPPER(C.CORREO)) = TRIM(UPPER(P_USUARIO));

        -----------------------------------------------------------------
        -- 2. VALIDAR QUE SEA EMPLEADO
        -----------------------------------------------------------------
        IF TRIM(UPPER(V_TIPO_USUARIO)) <> 'EMPLEADO' THEN
            P_RESULT := 'ERROR: SOLO EMPLEADOS PUEDEN INICIAR SESIÓN';
            RETURN;
        END IF;

        -----------------------------------------------------------------
        -- 3. RETORNAR JSON
        -----------------------------------------------------------------
        P_RESULT :=
            '{"ID_USUARIO": ' || V_ID_USUARIO ||
            ', "NOMBRE": "'   || V_NOMBRE ||
            '", "ROL": "'     || V_TIPO_USUARIO || '"}';

    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            P_RESULT := 'ERROR: USUARIO NO EXISTE';
        WHEN TOO_MANY_ROWS THEN
            P_RESULT := 'ERROR: CORREO DUPLICADO';
        WHEN OTHERS THEN
            P_RESULT := 'ERROR: ' || SQLERRM;
    END FIDE_USUARIOS_LOGIN_USUARIO_SP;

    PROCEDURE FIDE_FACTURAS_POR_EMPLEADO_SP(
    P_ID_EMPLEADO IN NUMBER,
    P_CURSOR OUT SYS_REFCURSOR
)
IS
BEGIN
    OPEN P_CURSOR FOR
        SELECT 
            F.FIDE_FACTURAS_ID_FACTURA_PK AS ID_FACTURA,
            F.FECHA_FACTURA,
            U.NOMBRE || ' ' || U.APELLIDO_PATERNO || ' ' || 
            U.APELLIDO_MATERNO AS EMPLEADO,
            F.TOTAL
        FROM FIDE_FACTURAS_TB F
        LEFT JOIN FIDE_USUARIOS_TB U
            ON F.FIDE_FACTURAS_ID_EMPLEADO_FK = U.FIDE_USUARIOS_ID_USUARIO_PK
        WHERE F.FIDE_FACTURAS_ID_EMPLEADO_FK = P_ID_EMPLEADO
          AND F.FIDE_FACTURAS_ID_ESTADO_FK = 1
        ORDER BY F.FECHA_FACTURA DESC
        FETCH FIRST 20 ROWS ONLY; 
END FIDE_FACTURAS_POR_EMPLEADO_SP;

            dir.DETALLE AS DETALLE_DIRECCION,
            e.NOMBRE_ESTADO,

            -- IDs necesarios para edición
            dir.FIDE_DIRECCIONES_ID_DIRECCION_PK AS ID_DIRECCION,
            p.FIDE_PROVINCIAS_ID_PROVINCIA_PK AS ID_PROVINCIA,
            c.FIDE_CANTONES_ID_CANTON_PK AS ID_CANTON,
            d.FIDE_DISTRITOS_ID_DISTRITO_PK AS ID_DISTRITO,

            dir.CODIGO_POSTAL AS CODIGO_POSTAL,
            s.FIDE_SUCURSALES_ID_ESTADO_FK AS ID_ESTADO

        FROM FIDE_SUCURSALES_TB s

        -- dirección (detalle, código postal, id distrito)
        JOIN FIDE_DIRECCIONES_TB dir 
            ON dir.FIDE_DIRECCIONES_ID_DIRECCION_PK = s.FIDE_SUCURSALES_ID_DIRECCION_FK

        -- distrito → cantón → provincia
        JOIN FIDE_DISTRITOS_TB d
            ON d.FIDE_DISTRITOS_ID_DISTRITO_PK = dir.FIDE_DIRECCIONES_ID_DISTRITO_FK

        JOIN FIDE_CANTONES_TB c
            ON c.FIDE_CANTONES_ID_CANTON_PK = d.FIDE_DISTRITOS_ID_CANTON_FK

        JOIN FIDE_PROVINCIAS_TB p
            ON p.FIDE_PROVINCIAS_ID_PROVINCIA_PK = c.FIDE_CANTONES_ID_PROVINCIA_FK

        -- estado sucursal
        JOIN FIDE_ESTADOS_TB e
            ON e.FIDE_ESTADOS_ID_ESTADO_PK = s.FIDE_SUCURSALES_ID_ESTADO_FK

        ORDER BY s.FIDE_SUCURSALES_ID_SUCURSAL_PK;
END FIDE_SUCURSALES_LISTAR_SP;

PROCEDURE FIDE_SUCURSALES_LISTAR_POR_PROVINCIA_SP(
    P_ID_PROVINCIA IN NUMBER,
    OUT_CURSOR OUT SYS_REFCURSOR
) AS
BEGIN
    OPEN OUT_CURSOR FOR
        SELECT 
            s.FIDE_SUCURSALES_ID_SUCURSAL_PK,
            s.NOMBRE_SUCURSAL,
            s.TELEFONO,
            s.FECHA_APERTURA,

            p.NOMBRE_PROVINCIA,
            c.NOMBRE_CANTON,
            d.NOMBRE_DISTRITO,

            dir.DETALLE AS DETALLE_DIRECCION,
            e.NOMBRE_ESTADO

        FROM FIDE_SUCURSALES_TB s

        -- DIRECCIÓN → DISTRITO
        JOIN FIDE_DIRECCIONES_TB dir
            ON dir.FIDE_DIRECCIONES_ID_DIRECCION_PK = s.FIDE_SUCURSALES_ID_DIRECCION_FK

        -- DISTRITO → CANTÓN → PROVINCIA
        JOIN FIDE_DISTRITOS_TB d
            ON d.FIDE_DISTRITOS_ID_DISTRITO_PK = dir.FIDE_DIRECCIONES_ID_DISTRITO_FK

        JOIN FIDE_CANTONES_TB c
            ON c.FIDE_CANTONES_ID_CANTON_PK = d.FIDE_DISTRITOS_ID_CANTON_FK

        JOIN FIDE_PROVINCIAS_TB p
            ON p.FIDE_PROVINCIAS_ID_PROVINCIA_PK = c.FIDE_CANTONES_ID_PROVINCIA_FK

        -- ESTADO DE LA SUCURSAL
        JOIN FIDE_ESTADOS_TB e
            ON e.FIDE_ESTADOS_ID_ESTADO_PK = s.FIDE_SUCURSALES_ID_ESTADO_FK

        WHERE p.FIDE_PROVINCIAS_ID_PROVINCIA_PK = P_ID_PROVINCIA;
END FIDE_SUCURSALES_LISTAR_POR_PROVINCIA_SP;

  -- =========================
  PROCEDURE FIDE_USUARIOS_LISTAR_TABLA_SP (
    OUT_CURSOR OUT SYS_REFCURSOR
  ) AS
  BEGIN
    OPEN OUT_CURSOR FOR
      SELECT
        u.FIDE_USUARIOS_ID_USUARIO_PK AS ID,
        u.NOMBRE || ' ' || u.APELLIDO_PATERNO || ' ' || NVL(u.APELLIDO_MATERNO,'') AS NOMBRE_COMPLETO,

        ON u.FIDE_USUARIOS_ID_ESTADO_FK = e.FIDE_ESTADOS_ID_ESTADO_PK
      ORDER BY u.FIDE_USUARIOS_ID_USUARIO_PK;
  END FIDE_USUARIOS_LISTAR_TABLA_SP;
